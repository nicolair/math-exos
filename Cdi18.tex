\begin{tiny}(Cdi18)\end{tiny} Appliquons le théorème du rang à
\begin{displaymath}
  g_{|\Im(f)} \in \mathcal{L}(\Im(f),G)
\end{displaymath}
On obtient
\begin{multline*}
  \rg(f) = \dim(\Im(f)) \\
  = \dim(\ker(g_{|\Im(f)})) + \dim(\Im(g_{|\Im(f)})) 
\end{multline*}
avec
\begin{multline*}
  \ker(g_{|\Im(f)}) = \ker(g) \cap \Im(f) \\ 
  \Im(g_{|\Im(f)}) = \Im(g\circ f)
\end{multline*}
