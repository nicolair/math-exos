\[
   D_x(A) = \left\lbrace |x-a|, a\in A\right\rbrace, \; d(x,A) = \inf D_x(A).
 \]

  \item Montrer que
  \[
    \forall (x,y) \in \R^2, \; \left| d(x,A) - d(y,A)\right| \leq | x - y|.
  \]

  \begin{multline*}
    d(x,A) = 0 \\

    \left\lbrace
    \begin{aligned}
      &(a_n)_{n \in \N}\rightarrow x \\

    \right. .
  \end{multline*}

\end{enumerate}
