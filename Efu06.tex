\begin{tiny}(Efu06)\end{tiny} 
\begin{enumerate}
  \item Montrer que 
\begin{displaymath}
\forall x \in ]-1,1[,\; \arctan \frac{x}{\sqrt{1-x^{2}}} = \arcsin x.
\end{displaymath}
  \item Pour quels $x$ l'équation
\begin{displaymath}
\arccos \frac{1-x}{1+x} + \arcsin \frac{2\sqrt{x}}{1+x} = \pi 
\end{displaymath}
est-elle définie? Résoudre en posant $x=\tan^2 \theta$.
\end{enumerate}
