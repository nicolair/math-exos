\begin{align*}
&\left\{
\begin{aligned}
(x^{2}+y^{2})\sin \frac{1}{x^{2}+y^{2}} & \text{ si } & (x,y)\neq (0,0) \\
0 & \text{ si } & (x,y)=(0,0)
\end{aligned}
\right.  \\
&\left\{
\begin{aligned}
\frac{x^{3}-y^{3}}{x^{2}+y^{2}} & \text{ si } & (x,y)\neq (0,0) \\
0 & \text{ si } & (x,y)=(0,0)
\end{aligned}
\right.  \\
&\left\{
\begin{aligned}
e^{\frac{1}{x^{2}+y^{2}-1}} & \text{ si } & x^{2}+y^{2}<1 \\
0 & \text{ si } & x^{2}+y^{2}\geq 1
\end{aligned}
\right.  \\
&\left\{
\begin{aligned}
x^{2} & \text{ si } & \left| x\right| >y \\
0 & \text{ si } & \left| x\right| \leq y
\end{aligned}
\right.
\end{align*}
