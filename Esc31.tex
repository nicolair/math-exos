\begin{tiny}(Esc31)\end{tiny}  Soit $a>0$ fixé.
\begin{enumerate}
 \item Vérifier que , pour tout $n\in \N^*$, 
\begin{displaymath}
 x+x^2+\cdots + x^n=a
\end{displaymath}
admet une unique solution dans $]0, +\infty[$ notée $x_n$.
\item Montrer que $\left( x_n\right) _{n\in \N^*}$ est décroissante. Préciser sa limite.
\end{enumerate}
