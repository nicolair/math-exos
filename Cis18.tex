\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{|l|l|l|l|l|} \hline
subdivision & $\left[0,1 \right[$ & $\left[1,2 \right[$ & $\left[2,3 \right[$ & $\left[3,4 \right[$ \\ \hline
valeurs & 0 & $\sin \frac{\pi}{4}$ & $\sin \frac{\pi}{2}$ & $\sin \frac{3\pi}{4}$ \\ \hline
\end{tabular}
\end{center}
