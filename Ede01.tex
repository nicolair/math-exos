\begin{align*}
& 3\sin 2x-2\sin 3x\;  (4)

&\sqrt{1+\cos x}\; (4) \\
&e^{\sin x}\; (3) 
&\sqrt{1+\tan x}\; (3) \\
&\frac{\sin x}{\ln (1+x)}\; (3) 
&\tan^{2}x^{2}\; (4)\\
&e^{\tan x}\; (4)
&\sqrt{1+\sqrt{1-x}}\; (2) \\
&\int_{x^3}^{x^2}\frac{dt}{\sqrt{t^4 + t^2 + 1}}\; (10)
\end{align*}
