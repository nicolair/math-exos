\begin{tiny}(Cvs24)\end{tiny} Si $A \cup B = B \cap C$,
\[
  \left. 
  \begin{aligned}
    A \cup B \subset B &\Rightarrow A \subset B \\
    A \cup B \subset C &\Rightarrow B \subset C
  \end{aligned}
  \right\rbrace \Rightarrow A \subset B \subset C.
\]
