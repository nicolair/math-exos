\begin{vmatrix}
1 & 1 & 1 & 1 \\
a & b & c & d \\
a^{2} & b^{2} & c^{2} & d^{2} \\
bcd & acd & abd & abc
\end{vmatrix},\hspace{0.5cm}
\begin{vmatrix}
a & a & a & a \\
a & b & b & b \\
a & b & c & c \\
a & b & c & d
\end{vmatrix}, \\

b+c & bc-1 & (b^{2}+1)(c^{2}+1) \\
c+a & ca-1 & (c^{2}+1)(a^{2}+1) \\
a+b & ab-1 & (a^{2}+1)(b^{2}+1)
\end{vmatrix}\\
\begin{vmatrix}
a+b     & b+c     & c+a \\
a^2+b^2 & b^2+c^2 & c^2+a^2 \\
a^3+b^3 & b^3+c^3 & c^3+a^3
\end{vmatrix}
\end{multline*}
