\begin{tiny}(Egs02)\end{tiny}
Soit $n\geq 3$.
\begin{enumerate}
 \item Montrer que les cycles de longueur $3$ engendrent le sous-groupe des permutations paires $\mathcal{A}_{n}$.
 \item Soit $a$, $b$, $c$ des éléments deux à deux distincts dans $\{3,\cdots,n\}$. Calculer
\begin{multline*}
 \begin{pmatrix}
  1&2&b
 \end{pmatrix}
\circ
 \begin{pmatrix}
  1&2&c
 \end{pmatrix}
\circ
 \begin{pmatrix}
  1&2&a
 \end{pmatrix} \\
\circ
 \begin{pmatrix}
  1&2&b
 \end{pmatrix}
\circ
 \begin{pmatrix}
  1&2&c
 \end{pmatrix}
\end{multline*}

Calculer
\begin{displaymath}
 \begin{pmatrix}
  1&2&c
 \end{pmatrix}
\circ
 \begin{pmatrix}
  1&2&b
 \end{pmatrix}
\circ
 \begin{pmatrix}
  1&2&b
 \end{pmatrix}
\end{displaymath}
\item  Montrer que les cycles
$\begin{pmatrix}
  1&2&i
 \end{pmatrix}
$
, pour $i$ entier dans $\left\{3,\cdots ,n\right\}$, engendrent $\mathcal{A}_{n}$. 
\end{enumerate}
