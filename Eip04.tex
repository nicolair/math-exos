\begin{align*}
\frac{1}{4x^2+4x+5},\; \frac{1}{x^2-4x+2} ,\; \frac{1}{5+3\cos x} \\
\frac{1}{x}\sqrt{\frac{x-1}{x+1}},\; \frac{x}{\sqrt{x^2+x+2}},\; \frac{1}{e^x+2e^{-x}}\\
\frac{x^2}{\sqrt{1-x^2}},\; \frac{\sin x +2\cos x}{\sin x -\cos x}
\end{align*}
