\begin{align*}
\frac{e^{\sin x}-e^{\tan x}}{\sin x-\tan x}; & &
\frac{(1+x)^{\frac{\ln x}{x}}-x}{x(x^{x}-1)};& & 
\left( \cos x\right) ^{\cotg x^{2}};\\
f(x) =x^{x};& & g(x)=x^{f(x)}; & & h(x)=x^{g(x)}; \\
\left( \frac{\sin x}{x}\right) ^{\frac{1}{\sin x}}; & &
\left( \ln (e+x)\right) ^{\frac{1}{x}}; & & 
\frac{x^{x^{x}}\ln x}{x^{x}-1};\\
